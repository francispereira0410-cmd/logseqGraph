  {
   :title [:h3 "📅 Today & Upcoming"]
   :query [:find (pull ?p [*])
           :in $ ?today
           :where
           [?p :block/name]
           [?p :block/journal? true]
           [?p :block/journal-day ?d]
           [(>= ?d ?today)]]
   :inputs [:today]
   :view (fn [rows]
         [:table
          [:thead
           [:tr
            [:th "Page"]
            [:th "Date"]
            [:th "Updated"]]]
          [:tbody
           (for [r (sort-by (fn [h] (get h :block/journal-day)) rows)]
             [:tr
              [:td [:a {:href (str "#/page/" (get r :block/name))} (get r :block/original-name)]]
              [:td (get r :block/journal-day)]
              [:td (let [updated (get r :block/updated-at)]
                     (if updated 
                         (subs (str updated) 0 10) 
                         "N/A"))]])]])
   :collapsed? false
  }
  #+END_QUERY
