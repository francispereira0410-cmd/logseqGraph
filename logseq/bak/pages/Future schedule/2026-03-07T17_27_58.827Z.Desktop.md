query-table:: true
#+BEGIN_QUERY
{:title [:h2 "📅 Upcoming Timeline (Ascending)"]
 :query [:find (pull ?b [*])
         :in $ ?today ?next
         :where
         [?b :block/refs ?journal]
         [?journal :block/journal-day ?d]
         [(>= ?d ?today)]
         [(<= ?d ?next)]]
 :inputs [:today :365d]
 :result-transform (fn [result]
                    (sort-by (fn [r] 
                               (get-in r [:block/refs 0 :block/journal-day])) 
                             result))
 :collapsed? false}
#+END_QUERY
